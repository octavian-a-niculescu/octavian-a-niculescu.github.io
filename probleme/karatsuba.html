Sa se implementeze o metoda eficienta de inmultire a doua numere in baza 10. 
int Power(int base, int exponent)
{
	int answer = 1;

	for (int power = 1; power <= exponent; power++)
		answer *= base;

	return answer;
}

int Split(int number, int half, char position)
{
	if (position == 'L')
	{
		return number / Power(10, half);
	}
	if (position == 'R')
	{
		return number % Power(10, half);
	}
}

int SizeOf(int number)
{
	int sizeOfNumber = 0;

	do
	{
		sizeOfNumber++;
		number /= 10;

	} while (number);

	return sizeOfNumber;
}

int Karatsuba(int number1, int number2)
{
	if (number1 < 10 || number2 < 10)
	{
		return number1 * number2;
	}

	int min = std::min(SizeOf(number1), SizeOf(number2));

	int leftOfNumber1 = Split(number1, min / 2, 'L');
	int rightOfNumber1 = Split(number1, min / 2, 'R');
	int leftOfNumber2 = Split(number2, min / 2, 'L');
	int rightOfNumber2 = Split(number2, min / 2, 'R');

	


	int multiplication1 = Karatsuba(leftOfNumber1, leftOfNumber2);
	int multiplication2 = Karatsuba((rightOfNumber1 - leftOfNumber1), (leftOfNumber2 - rightOfNumber2));
	//int multiplication2 = Karatsuba((rightOfNumber1 + leftOfNumber1), (leftOfNumber2 + rightOfNumber2));
	int multiplication3 = Karatsuba(rightOfNumber1, rightOfNumber2);


	return multiplication1 * Power(10, min / 2 * 2) + ((multiplication2 + multiplication1 + multiplication3) * Power(10, min / 2)) + multiplication3;
	//return multiplication1 * Power(10, min / 2 * 2) + ((multiplication2 - multiplication1 - multiplication3) * Power(10, min / 2)) + multiplication3;


}


int main()
{
	int number1, number2;

	std::cin >> number1 >> number2;
	std::cout << Karatsuba(number1, number2);
}
Explicatie rezolvare: Functia Karatsuba este o functie recursiva care la fiecare pas sectioneaza numerele de inmultit in cate 2 alte numere de lungimi aproximativ egale.Functia se reapeleaza de 3 ori avand ca parametri combinatii de cele 4 numere obtinute respectand formula de mai sus.
